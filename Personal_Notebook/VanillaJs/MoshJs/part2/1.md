# OOP

```js
// Factory Function
function createCircle(radius) {
  return {
    radius,
    draw: function() {
      console.log('draw...')
    }
  }
}

const circle = createCircle(1);
circle.draw();

// Constructor Function
function Circle(radius) {
  this.radius = radius;
  this.draw = () => {
    console.log('drawing....')
  }
}

const another = new Circle(1);
another.draw();
```
